2. FUNCAO HELPER

   Criada funcao formatCNPJ():
   
   ```javascript
   const formatCNPJ = (cnpj) => {
     if (!cnpj) return ''
     // Remove caracteres nao numericos
     const cleaned = cnpj.replace(/\D/g, '')
     // Aplica mascara XX.XXX.XXX/XXXX-XX
     if (cleaned.length === 14) {
       return cleaned.replace(
         /^(\d{2})(\d{3})(\d{3})(\d{4})(\d{2})$/,
         '$1.$2.$3/$4-$5'
       )
     }
     return cnpj
   }
   ```
